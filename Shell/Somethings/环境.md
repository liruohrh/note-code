    echo "Skip $cmd: executable not found"
    continue

    if "$cmd" --version &>/dev/null || \
       "$cmd" --help &>/dev/null || \
       "$cmd" -h &>/dev/null; then
      	echo "cmd usable: $cmd"
	continue
    fi
  fi
