
# pprof

```go

import (
	_ "net/http/pprof"
)

func main() {  
    go func() {  
	   //如果没有启动默认的http server，则启动一下
       if err := http.ListenAndServe(":6060", nil); err != nil && errors.Is(err, http.ErrServerClosed) {  
          log.Printf("pprof server error: %s", err)  
       }  
    }()  
}
```
